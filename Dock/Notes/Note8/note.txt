public class Validator<T> {
	public T validate(Untrusted<T> untrusted) {
		if (!innerValidate(untrusted.value) {
			throw new ValidationException("");
		}

		return untrusted.value;
	}

	protected abstract boolean innerValidate(T value);
}

public <R> R process(String userId, Remote.Function<Cart, R> cartFunction) {
	return carts.invoke(userId, entry -> {
		Cart cart = entry.getValue();
		R result = cartFunction.apply(cart);
		entry.setValue(cart);

		return result;
	});
}
